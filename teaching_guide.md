WITH attrition_stats AS (

           COUNT(CASE WHEN w.attrition = 'Yes' THEN 1 END) * 100.0 / COUNT(*) as attrition_rate
    FROM employee_work_history w

SELECT * FROM attrition_stats
UNION ALL
SELECT 'OT: ' || over_time, attrition_rate, NULL, NULL 
FROM overtime_analysis;

#### **Aggregate Window Functions (20 ‡∏ô‡∏≤‡∏ó‡∏µ)**

                                   ORDER BY c.monthly_income) as running_total,
       AVG(c.monthly_income) OVER (PARTITION BY d.department_id) as dept_avg_salary,
       c.monthly_income - AVG(c.monthly_income) OVER (PARTITION BY d.department_id) as diff_from_avg
FROM employees e
JOIN departments d ON e.department_id = d.department_id
JOIN employee_compensation c ON e.employee_id = c.employee_id
ORDER BY d.department_name, c.monthly_income;
```

#### **Integration: CTEs + Window Functions (10 ‡∏ô‡∏≤‡∏ó‡∏µ)**

-- Complex Analysis: ‡∏´‡∏≤ High-Risk Employees
WITH employee_metrics AS (
    SELECT e.employee_id,
           c.monthly_income,
           w.attrition,
           w.over_time,
           s.job_satisfaction,
           w.years_at_company,
           NTILE(4) OVER (ORDER BY c.monthly_income) as income_quartile,
           AVG(s.job_satisfaction) OVER () as avg_satisfaction
    FROM employees e
    JOIN employee_compensation c ON e.employee_id = c.employee_id
    JOIN employee_work_history w ON e.employee_id = w.employee_id
    JOIN employee_satisfaction s ON e.employee_id = s.employee_id
)
SELECT employee_id,
       CASE 
           WHEN over_time = 'Yes' 
                AND income_quartile <= 2 
                AND job_satisfaction < avg_satisfaction 
                AND years_at_company < 5 
           THEN 'High Risk'
           WHEN over_time = 'Yes' OR income_quartile = 1 
           THEN 'Medium Risk'
           ELSE 'Low Risk'
       END as risk_level,
       monthly_income,
       job_satisfaction,
       years_at_company
FROM employee_metrics
WHERE attrition = 'No'
ORDER BY risk_level;
```
